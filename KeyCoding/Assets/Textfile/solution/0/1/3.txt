#include<stdio.h>
int main(){
		int i, n;
		printf("원하시는 단을 입력해주세요. : ");
		scanf("%d", &n);
		for(i=1;i<=9;i++){
			printf("%d * %d = %d", n, i, n*i);
		}
		return 0;
}

/*
for문이란? 정해진 횟수나 특정한 번호부터 번호까지 반복시키고 싶을 때 사용합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    for(변수명=초기값;반복할 조건;증/감)
-------------------------------------------------------------
*/
