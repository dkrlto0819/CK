class Program{
		static void Main(string[] args){
			int[] a=new int[10]{45, 6, 2, 90, 1, 34, 27, 30, 2, 10};
			int temp;
	
			for(int i=0;i<9;i++){
				for(j=0;j<9;j++){
					if(a[j]>a[j+1]){
						temp=a[j];
						a[j]=a[j+1];
						a[j+1]=temp;
					}
	//j번째 값과 j+1번째 값을 비교합니다.
	//j번째 값이 더 클 경우 값을 교환해줍니다.
				}
			}
			for(i=0;i<10;i++){
				Console.WriteLine(a[i]);
			}
		}

}

/*
for문이란? 정해진 횟수나 특정한 번호부터 번호까지 반복시키고 싶을 때 사용합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    for(변수명=초기값;반복할 조건;증/감)
-------------------------------------------------------------

배열이란? 같은 타입의 자료가 여러 개 열거된 자료를 말합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    자료형 배열이름=new 자료형[자료 개수]{저장할 자료들};
-------------------------------------------------------------

if문이란? 조건에 따라 실행할 문장을 나누는 함수입니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    if(조건){ }
    else if(조건){ }
    else { }
-------------------------------------------------------------
*/
