class program{
		static void Main(String[] args){
			int i, sum=0;
			for(i=1;i<=100;i++){
			//i는 1부터 100까지 증가
				sum += i;
			//sum은 계속 i 만큼 더해짐
			}
			Console.WriteLine(sum);
		}
}

/*
for문이란? 정해진 횟수나 특정한 번호부터 번호까지 반복시키고 싶을 때 사용합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    for(변수명=초기값;반복할 조건;증/감)
-------------------------------------------------------------
*/
