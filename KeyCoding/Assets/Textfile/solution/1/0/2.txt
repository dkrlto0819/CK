class Program{
		static void Main(string[] args){
			int[] a=new int[5] {3, 6, 7, 81, 9};
			// 데이터가 5개 있는 배열
			for(int i=0;i<5;i++) Console.WriteLine(a[i]);
		}
}

/*
for문이란? 정해진 횟수나 특정한 번호부터 번호까지 반복시키고 싶을 때 사용합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
    for(변수명=초기값;반복할 조건;증/감)
-------------------------------------------------------------

배열이란? 같은 타입의 자료가 여러 개 열거된 자료를 말합니다.
기본 형태는 아래와 같습니다.
-------------------------------------------------------------
     자료형 배열이름=new 자료형[자료 개수]{저장할 자료들};
-------------------------------------------------------------
*/
