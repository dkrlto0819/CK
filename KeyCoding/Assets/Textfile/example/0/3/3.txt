#include<stdio.h>
int main(){
	int j, n, temp;
	int a[100];
	printf("몇 개의 수를 입력받을 것인지 입력하시오. : ");
	scanf("%d", &n);

	for(int i=0;i<n;i++){
		printf("수를 입력하시오. : ");
		scanf("%d", &a[i]);
	}
	
	for(i=1;i<n;i++){
		temp=a[i];
		j=i-1;
		while(j>=0 && a[j]>temp){
			a[j+1]=a[j];
			j=j-1;
		}
		a[j+1]=temp;
	}
	for(i=0;i<n;i++){
		printf(“%d ”, a[i]);
	}
	return 0;
}
