#include<stdio.h>
int main(){
	int a[100];
	int n, temp;
	printf("몇 개의 수를 입력받을 것인지 입력하시오. : ");
	scanf("%d", &n);

	for(int i=0;i<n;i++){
		printf("수를 입력하시오. : ");
		scanf("%d", &a[i]);
	}

	for(int i=0;i<n-1;i++){
		for(j=0;j<n-1;j++){
			if(a[j]>a[j+1]){
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	for(i=0;i<n;i++){
		printf("%d ", a[i]);
		printf("\n");
	}
}
